Immediately Invoked Function Expression

//iifes function
(function imi(x,y) {
var xy = x*y;
document.write(xy);
})(10,20);
//এখানে সাধারন ভাবে ফাংশন তৈরি করা হয়েছে, যা ইমেডিয়েট কলের জন্য সম্পূর্ণ ফংশনকে ()এর মধ্যে রাখা হয়েছে। এবং শেষে ( )এর মধ্যে প্যারামিটার x,y এর মান বসানো হয়েছে।
