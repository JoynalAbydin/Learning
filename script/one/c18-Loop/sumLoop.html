এখানে এই প্রোগ্রামে প্রত্যেকটি সংখ্যাকে একবার একবার করে যোগ করবে যার কারণে বড় ধরনের সংখ্যা এর যোগফল বের করতে চাইলে এটি সাধারণভাবে অনেক সময় নেবে অনেক সময় পেজটির লোড হবে না।।।

একই সাথে এই প্রগ্রামটি সুত্রের মাধ্যমে করা হয়েছে যার কারনে এখানে বরো সংখ্যা দিলেও এটা সহজে উত্তর হিসাব করুন


For Loop Functions

var x = parseInt(prompt("Enter starting number: "));
var y = parseInt(prompt("Enter ending number"));

var sum = 0;
for (var i = 1; i <= 100; i = i + 1) //এখানে i এর মান ১ ধরে যোগ কার হেয়েছে
{ sum = sum + i; }
document.write(sum);

var jog = 0;
for (var j = x; j <= y; j = j + 1) //এখানে প্রথম (Prompt ইনপুট) x এর মান var = j এর মাঝে রাখা হয়েছে। j যতক্ষন y এর সমান না হবে ততক্ষ এই লুপটা চলবে।
{ jog = jog + j; }
document.write("Your Result is = "+jog);
