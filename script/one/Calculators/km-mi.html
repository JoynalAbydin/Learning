Kilometer Converter

এখানে কিলোমিটার থেকে মাইল কনভার্ট একই সাথে মাইল থেকে কিলোমিটার কনভার্ট হরা যাবে

এই পেজে প্রবেশের পর আপনাকে ইনপুট দেওয়ার ঘর আসবে। যেখানে আপনি যেকোনো সংখ্যা ইনপুট দিলে এটি Kilometre To Mile = .... Mile To Kilometre = .... দুই ভাবেই আপনার রেজাল্ট দেখাবে।

আপনি যদি ইনপুট দেওয়ার অপশন না পান তাহলে পেজটি রিফ্রেশ করুন তাহলে আবার ইনপুট দেওয়ার অপশন আসবে।

আপনার প্রদত্ত সংখ্যার ফলাফল

সুত্র এবং সোর্স কোড দেখুন

var km = parseFloat(prompt("Enter km or ml: "));
var k2m = km*0.62137119; //কিলোমিটার থেকে মাইল বের করার সুত্র
document.write("Kilometre To Mile = "+ k2m.toFixed(2)+"
");
var m2k = km*1.609344; //মাইল থেকে কিলোমিটার বের কার সুত্র
document.write("Mile To Kilometre = "+m2k.toFixed(2));
